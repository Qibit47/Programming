**3.36**
```C++
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int ia1[10], ia2[10];
	int *e1 = &ia1[10], *e2 = &ia2[10];
	for (int *p1 = ia1, *p2 = ia2; p1 != e1; p1++)
		if (*p1 != *p2++) break;
	cout << (p1 == e1 && p2 == e2) ? "Not" : "" << "Same Array" << endl;
	
	vector<int> ivec1, ivec2;
	cout << (ivec1 == ivec2 ? "Not" : "") << "Same Vector" << endl;
	return 0;
}
```
